代理代码 (Node.js):
npm install express cors node-fetch const app = require('express')(); const cors = require('cors'); app.use(cors({origin: '*'})); app.use(require('express').json({limit: '50mb'})); app.all('/proxy/*', async (req, res) => { const url = 'https://api.toapis.com/v1' + req.path.replace('/proxy',''); const r = await fetch(url, { method: req.method, headers: {Authorization: req.headers.authorization, 'Content-Type': 'application/json'}, body: JSON.stringify(req.body) }); res.json(await r.json()); }); app.listen(3000);
